import { Component } from '@angular/core';
import { Store } from '@ngrx/store';
import { Observable } from 'rxjs';

import { getMonthlyAccountBalanceMovements } from '../../../application/state/analysis.selectors';
import { AnalysisState } from '../../../domain/state/analysis.state';

import { Movement } from '@features/movements/domain/models/movement.model';

@Component({
    selector: 'app-analysis-monthly-movements',
    templateUrl: './monthly-movements.component.html',
    styleUrls: ['./monthly-movements.component.scss'],
})

/**
 * Analysys monthly movements component
 */
export class AnalysisMonthlyMovementsComponent {
    public monthyMovements$: Observable<Movement[]> = this.store.select(getMonthlyAccountBalanceMovements);

    public movementsCollapsed = false;

    constructor(private readonly store: Store<AnalysisState>) {

    }
}

<div class="row">
	<div class="col-sm-12">
		<div class="white-box">
			<h3 class="box-title">
				Movements
				<div class="float-end" role="button" (click)="movementsCollapsed = !movementsCollapsed">
					<i [class]="movementsCollapsed ? 'fa fa-chevron-down' : 'fa fa-chevron-up'"></i>
				</div>
			</h3>
			<div class="table-responsive" #collapse="ngbCollapse" [(ngbCollapse)]="movementsCollapsed">
				<table class="table text-nowrap table-striped">
					<thead>
						<tr>
							<th class="border-top-0">#</th>
							<th class="border-top-0">Name</th>
							<th class="border-top-0">Amount</th>
							<th class="border-top-0">Date</th>
							<th class="border-top-0">Category</th>
							<th class="border-top-0">Account</th>
						</tr>
					</thead>
					<tbody>
						<tr
							*ngFor="let movement of monthyMovements$ | async"
							[ngClass]="{ expense: movement.type === 'expense', income: movement.type === 'income' }"
						>
							<td>{{ movement.id }}</td>
							<td>{{ movement.name }}</td>
							<td>{{ movement.amount | currency }}</td>
							<td>{{ movement.date }}</td>
							<td class="category">
								<i class="fa {{ movement.category?.icon }}"></i>
								{{ movement.category?.name }}
							</td>
							<td>{{ movement.account?.name }}</td>
						</tr>
					</tbody>
				</table>
			</div>
		</div>
	</div>
</div>
